Reading Sensor Data with Python

Code examples for interfacing with various sensors.

Ultrasonic Sensor 
import RPi.GPIO as GPIO
import time

GPIO.setmode(GPIO.BCM)
TRIG = 23
ECHO = 24

GPIO.setup(TRIG, GPIO.OUT)
GPIO.setup(ECHO, GPIO.IN)

def get_distance():
    GPIO.output(TRIG, True)
    time.sleep(0.00001)
    GPIO.output(TRIG, False)
    
    while GPIO.input(ECHO) == 0:
        pulse_start = time.time()
    
    while GPIO.input(ECHO) == 1:
        pulse_end = time.time()
    
    pulse_duration = pulse_end - pulse_start
    distance = pulse_duration * 17150
    distance = round(distance, 2)
    
    return distance

try:
    while True:
        dist = get_distance()
        print(f"Distance: {dist} cm")
        time.sleep(1)
except KeyboardInterrupt:
    GPIO.cleanup()

Controlling Motors with Python

Code examples for DC motor and servo control.

DC Motor Control 
from gpiozero import Motor
from time import sleep

motor = Motor(forward=17, backward=18)

def motor_control(speed, duration):
    if speed > 0:
        motor.forward(speed)
    elif speed < 0:
        motor.backward(abs(speed))
    else:
        motor.stop()
    
    sleep(duration)
    motor.stop()

# Example usage
motor_control(0.5, 2)  # Forward at 50% speed for 2 seconds
motor_control(-0.3, 1) # Backward at 30% speed for 1 second

Robotics Algorithms

Common robotics algorithms implemented in Python.

PID Controller 
class PIDController:
    def __init__(self, Kp, Ki, Kd):
        self.Kp = Kp
        self.Ki = Ki
        self.Kd = Kd
        self.prev_error = 0
        self.integral = 0
    
    def compute(self, setpoint, measured_value, dt):
        error = setpoint - measured_value
        self.integral += error * dt
        derivative = (error - self.prev_error) / dt
        output = self.Kp * error + self.Ki * self.integral + self.Kd * derivative
        self.prev_error = error
        return output

# Example usage
pid = PIDController(Kp=1.0, Ki=0.1, Kd=0.01)
setpoint = 10  # Target value
measured_value = 5  # Current sensor reading
dt = 0.1  # Time step

control_signal = pid.compute(setpoint, measured_value, dt)
print(f"Control signal: {control_signal}")
